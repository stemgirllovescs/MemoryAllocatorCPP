285pointer allocate (size_type s) {
286 int size_in_bytes = s * 8; // Object size is 8 bytes
287
288 for (iterator it = begin(); it != end(); ++it) {
289 int& block_size = *it;
290 if (block_size > 0 && block_size >= size_in_bytes) {
291 int original_size = block_size;
292 int remaining = original_size - size_in_bytes - 8; // Remaining data size after allocating and adding end sentinel
293
294 if (remaining >= static_cast<int>(8)){
295 // Split the block
296 block_size = -size_in_bytes;
297 (*this)[it._i + 4 + size_in_bytes] = -size_in_bytes; // End sentinel for allocated block
298 // Create new free block
299 int new_block_index = it._i + 8 + size_in_bytes;
300 (*this)[new_block_index] = remaining;
301 (*this)[new_block_index + 4 + remaining] = remaining;
302 } else {
303 // Do not split, allocate entire block
304 block_size = -original_size;
305 (*this)[it._i + 4 + original_size] = -original_size;
306 }
307 assert(valid());
308 return reinterpret_cast<pointer>(&a[it._i +4]);
309 }
310 }
311 throw std::bad_alloc();
312}

355void deallocate(pointer p, size_type) {
356 int index = reinterpret_cast<char*>(p) - a - 4;
357 if (index < 0 || index >= static_cast<int>(N)) {
358 throw std::invalid_argument("Invalid pointer");
359 }
360
361 int& block_size = (*this)[index];
362 if (block_size >= 0) {
363 throw std::invalid_argument("Block is already free");
364 }
365
366 // Mark block as free
367 int size = -block_size;
368 block_size = size;
369 (*this)[index + 4 + size] = size;
370
371 // Coalesce with the next block if it is free
372 int next_index = index + size + 8;
373 if (next_index < static_cast<int>(N)) {
374 int& next_block_size = (*this)[next_index];
375 if (next_block_size > 0) {
376 // Update size to include the next free block and the header/footer
377 size += next_block_size + 8;
378 (*this)[index] = size; // Update start sentinel
379 (*this)[index + 4 + size] = size; // Update end sentinel
380 block_size = size; // Update block_size after coalescing
381 }
382 }
383
384 // Coalesce with the previous block if it is free
385 if (index > 0) {
386 int prev_end_index = index - 4;
387 int prev_size = (*this)[prev_end_index];
388 if (prev_size > 0) {
389 int prev_index = index - prev_size - 8;
390 // Update size to include the previous free block and the header/footer
391 size += prev_size + 8;
392 (*this)[prev_index] = size; // Update start sentinel
393 (*this)[prev_index + 4 + size] = size; // Update end sentinel
394 index = prev_index; // Update index to new start
395 block_size = size; // Update block_size to new size
396 }
397 }
398
399 assert(valid());
400}
